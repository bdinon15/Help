--DROP TABLES--

DROP TABLE BKD_WHISTLEBLOWER CASCADE CONSTRAINTS;
DROP TABLE BKD_LEAK CASCADE CONSTRAINTS;
DROP TABLE BKD_LAWSUIT CASCADE CONSTRAINTS;
DROP TABLE BKD_IMPACT CASCADE CONSTRAINTS;
DROP TABLE BKD_FINANCIAL CASCADE CONSTRAINTS; 
DROP TABLE BKD_HEALTH CASCADE CONSTRAINTS; 
DROP TABLE BKD_NATIONAL_SECURITY CASCADE CONSTRAINTS; 

--CREATE TABLES--

CREATE TABLE BKD_WHISTLEBLOWER(
NAME_ID VARCHAR2(30),
SEX_ID VARCHAR2(7),
AGE_ID NUMBER,
BIRTH_PLACE VARCHAR2(20),
PROFESSION_NAME VARCHAR2(20));

CREATE TABLE BKD_LAWSUIT(
LAWSUIT_ID VARCHAR2(20),
LAWSUIT_DATE DATE,
JAIL_TIME VARCHAR2(10),   
LAW_ACTION VARCHAR2(20)
);


CREATE TABLE BKD_WHISTLEBLOWER_LAWSUIT(
NAME_ID VARCHAR2(20),
LAWSUIT_ID VARCHAR2(20)
); 

CREATE TABLE BKD_LEAK(
LEAK_ID VARCHAR2(15),
LEAK_FMT VARCHAR2(15), -- METHOD TO HACK
SOURCE_ID VARCHAR2(15),  --WHERE IT CAME FROM + WHO REPORTED IT FIRST.
LEAK_DATE DATE,
ATTACK_TYPE VARCHAR2(20),
FILES_LEAKED NUMBER);

CREATE TABLE BKD_IMPACT(
AFFECTED_PARTY VARCHAR2(50),
AFFECTED_COUNTRIES VARCHAR2(100),
AFFECTED_LAWS VARCHAR2 (50),
AFFECTED_COMPANY VARCHAR2(20),
DEATH_TOLL NUMBER
);

CREATE TABLE BKD_FINANCIAL(  
FINANCIAL_BUREAUCRACY VARCHAR2(30),
INVOLVED_STOCKS VARCHAR2(100),
CURRENCY_LOST NUMBER
);
CREATE TABLE BKD_HEALTH(        
INVOLVED_STATES VARCHAR2(100),
HEALTH_BUREAUCRACY VARCHAR2(30),
ORIGIN_ID VARCHAR2(20)  --WHERE THE INCIDENT HAPPENED
);
--
CREATE TABLE BKD_NATIONAL_SECURITY(     -- MOST LIKELY GOING TO ADD MORE ATTRIBUTES AS I COME ACROSS NEW INFORMATION WHEN ADDING INSERT STATEMENTS. 
SECURITY_BUREAUCRACY VARCHAR2(30),
INVOLVED_COUNTRIES VARCHAR2(100)
);

--PK--
ALTER TABLE BKD_WHISTLEBLOWER
ADD CONSTRAINT NAME_ID PRIMARY KEY (NAME_ID);

ALTER TABLE BKD_LEAK
ADD CONSTRAINT LEAK_ID PRIMARY KEY (LEAK_ID);

ALTER TABLE BKD_LAWSUIT
ADD CONSTRAINT LAWSUIT_ID PRIMARY KEY (LAWSUIT_ID);

--FK--
ALTER TABLE BKD_IMPACT
ADD CONSTRAINT LEAK_IMPACT_FK FOREIGN KEY (AFFECTED_PARTY) REFERENCES BKD_LEAK (LEAK_ID);

ALTER TABLE BKD_WHISTLEBLOWER_LAWSUIT
ADD CONSTRAINT FK_WHISTLENAME FOREIGN KEY (NAME_ID) REFERENCES WHISTLEBLOWER (NAME_ID);

ALTER TABLE BKD_WHISTLEBLOWER_LAWSUIT
ADD CONSTRAINT FK_LAWSUIT FOREIGN KEY (LAWSUIT_ID) REFERENCES WHISTLEBLOWER (LAWSUIT_ID);
--SUBTYPES
ALTER TABLE BKD_HEALTH
ADD CONSTRAINT HEALTH_FK FOREIGN KEY (HEALTH_BUREAUCRACY) REFERENCES BKD_LEAK (LEAK_ID);

ALTER TABLE BKD_FINANCIAL
ADD CONSTRAINT FINANCIAL_FK FOREIGN KEY (FINANCIAL_BUREAUCRACY) REFERENCES BKD_LEAK (LEAK_ID);

ALTER TABLE BKD_NATIONAL_SECURITY
ADD CONSTRAINT SECURITY_FK FOREIGN KEY (SECURITY_BUREAUCRACY) REFERENCES BKD_LEAK (LEAK_ID);
